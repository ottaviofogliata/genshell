146 int rc = gs_strbuf_append_char(&buf, ch);
147 if (rc != GS_OK) {
148 gs_strbuf_dispose(&buf);
149 return rc;
150 }
151 literal_next = false;
152 continue;
153 }
154 if (ch == GS_LITERAL_SENTINEL) {
155 literal_next = true;
156 continue;
157 }

164 if (rc != GS_OK) {

168 continue;

211 }
212 if (isalnum((unsigned char)*p) || *p == '_') {
213 const char *start = p;
214 while (isalnum((unsigned char)*p) || *p == '_') {
215 ++p;
216 }
217 size_t len = (size_t)(p - start);
218 int rc = append_parameter(&buf, shell, start, len);
219 if (rc != GS_OK) {

224 }

226 if (rc != GS_OK) {
227 gs_strbuf_dispose(&buf);
228 return rc;
229 }
230 continue;
231 }
232 int rc = gs_strbuf_append_char(&buf, ch);
233 if (rc != GS_OK) {
234 gs_strbuf_dispose(&buf);
235 return rc;
236 }
237 }
238

489static int execute_pipeline_processes(struct gs_shell *shell, gs_prepared_command *cmds, size_t count) {
490 pid_t *pids = (pid_t *)calloc(count, sizeof(pid_t));
491 if (!pids) {
492 return GS_ERR_ALLOC;
493 }
494 int prev_read = -1;
495 int status_result = 0;
496
497 for (size_t i = 0; i < count; ++i) {
498 int pipefd[2] = {-1, -1};
499 if (i + 1 < count) {
500 if (pipe(pipefd) < 0) {
501 fprintf(stderr, "genshell: pipe failed: %s\n", strerror(errno));
502 if (prev_read >= 0) {
503 close(prev_read);
504 prev_read = -1;
505 }
506 status_result = GS_ERR_EXEC;
507 goto cleanup;
508 }
509 }
510
511 pid_t pid = fork();
512 if (pid < 0) {
513 fprintf(stderr, "genshell: fork failed: %s\n", strerror(errno));
514 if (pipefd[0] >= 0) {
515 close(pipefd[0]);
516 close(pipefd[1]);
517 }
518 if (prev_read >= 0) {
519 close(prev_read);
520 prev_read = -1;
521 }
522 status_result = GS_ERR_EXEC;
523 goto cleanup;
524 }
525
526 if (pid == 0) {
527 if (prev_read >= 0 && dup2(prev_read, STDIN_FILENO) < 0) {
528 fprintf(stderr, "genshell: dup2 failed: %s\n", strerror(errno));
529 _exit(1);
530 }
531 if (pipefd[1] >= 0 && dup2(pipefd[1], STDOUT_FILENO) < 0) {
532 fprintf(stderr, "genshell: dup2 failed: %s\n", strerror(errno));
533 _exit(1);
534 }
535
536 if (pipefd[0] >= 0) {
537 close(pipefd[0]);
538 }
539 if (pipefd[1] >= 0) {
540 close(pipefd[1]);
541 }
542 if (prev_read >= 0) {
543 close(prev_read);
544 }
545
546 if (cmds[i].builtin) {
547 execute_child_builtin(shell, &cmds[i]);
548 } else {
549 execute_child_external(&cmds[i]);
550 }
551 _exit(1); /* should not reach */
552 }
553
554 pids[i] = pid;
555
556 if (prev_read >= 0) {
557 close(prev_read);
558 }
559 if (pipefd[1] >= 0) {
560 close(pipefd[1]);
561 }
562 prev_read = pipefd[0];
563 }
564
565 if (prev_read >= 0) {
566 close(prev_read);
567 }
568
569 for (size_t i = 0; i < count; ++i) {
570 int wstatus = 0;
571 if (waitpid(pids[i], &wstatus, 0) < 0) {
572 status_result = GS_ERR_EXEC;
573 continue;
574 }
575 if (i == count - 1) {
576 if (WIFEXITED(wstatus)) {
577 status_result = WEXITSTATUS(wstatus);
578 } else if (WIFSIGNALED(wstatus)) {
579 status_result = 128 + WTERMSIG(wstatus);
580 }
581 }
582 }
583
584cleanup:
585 free(pids);
586 return status_result;
587}
